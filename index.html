function toggleArabic() {

  const ar = document.getElementById('arabic');

  ar.style.display = ar.style.display === 'none' ? 'block' : 'none';

}


async function loadCitation() {

  const res = await fetch('/api/citations?ref=Q96:1-5');

  const data = await res.json();

  document.getElementById('citation').textContent =

    JSON.stringify(data, null, 2);

}
